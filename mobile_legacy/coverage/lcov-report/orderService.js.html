/**
 * Validate that no items in the cart belong to a closed section.
 */
export async function validateCartItems(items) {
    // items is array of { id, ... }
    const itemIds = items.map(item => item.id);
 
    // We need 'Query' from sdk
    const { Query } = require('react-native-appwrite');
 
    const response = await databases.listDocuments(
        DATABASE_ID,
        COLLECTIONS.MENU_ITEMS,
        [Query.equal('$id', itemIds)]
    );
 
    const closedItems = response.documents.filter(item => item.sectionClosed);
 
    if (closedItems.length > 0) {
        const closedNames = closedItems.map(item => item.name).join(', ');
        const closedSections = [...new Set(closedItems.map(item => item.category))].join(', ');
        return { valid: false, message: `The following section(s) are now closed: ${closedSections}\n\nAffected items: ${closedNames}` };
    }
 
    return { valid: true };
}
 
/**
 * [ADMIN] Get all pending payments for verification
 */
export async function getPendingPayments() {
    console.log('üì¶ OrderService: Fetching pending payments...');
    return await databases.listDocuments(
        DATABASE_ID,
        COLLECTIONS.ORDERS,
        [
            Query.equal('status', 'pending_verification'),
            Query.orderDesc('$createdAt')
        ]
    );
}
 
/**
 * [ADMIN] Get All Orders with optional status filter
 */
export async function getOrders(status = null) {
    const queries = [Query.orderDesc('$createdAt')];
    if (status) {
        queries.push(Query.equal('status', status));
    }
 
    return await databases.listDocuments(
        DATABASE_ID,
        COLLECTIONS.ORDERS,
        queries
    );
}

/**
 * Get Payment Proof URL
 */
export function getPaymentProofUrl(fileId) {
    if (!fileId) return null;
    return storage.getFileView(BUCKET_ID, fileId).href;
}
 
/**
 * [ADMIN] Get Order by OTP
 */
export async function getOrderByOtp(otp) {
    const response = await databases.listDocuments(
        DATABASE_ID,
        COLLECTIONS.ORDERS,
        [
            Query.equal('otp', otp),
            Query.notEqual('status', 'collected')
        ]
    );
    return response.documents[0] || null;
}
